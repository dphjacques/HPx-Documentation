    Additions to database


    RATES
    pyrite
    -mybasic
    -start
     #1 REM parm(1) k_0 mol/dmï¿½/s
     mole = 0
     if SR("pyrite") < 1 then 
         MolarVolume = phase_vm("pyrite")
         phi0 = m0()*MolarVolume # initial bulk volume
         phi = m()*MolarVolume
         k_red = (phi/phi0)^(2/3)
         k = k_red*parm(1)/conv_t("s") # to convert from time units s to the time units of the HYDRUS GUI
         rate = k*(1 - SR("pyrite"))
         mole = rate*time()
         if (mole > m()) then mole = m()
     endif
     save(mole)
    -end
